create table tw_price_info (
	id INT NOT NULL AUTO_INCREMENT,
	date DATE DEFAULT (CURRENT_DATE),
	stock_id VARCHAR(20) NOT NULL,
	price FLOAT NOT NULL,   
	price_open FLOAT NOT NULL,
	high FLOAT NOT NULL,
	low FLOAT NOT NULL,
	volume INT NOT NULL,
	market_price FLOAT NOT NULL,
	PRIMARY KEY (id)
);
