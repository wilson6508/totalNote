package test0904;

public class Ex03 {

	public static void main(String[] args) {
						
		int y = 8; //第一排*號前有幾個空白		
		int x = 5; //印出幾排		
		StringBuffer sb = new StringBuffer("*");
		
		for(int j=1;j<=y;j++)
		{				
			sb.insert(0, " ");				
		}
		
		for(int j=1;j<=x;j++)
		{
			System.out.println(sb);
			sb.deleteCharAt(0);
			sb.append("*");
		}					
	}
}
