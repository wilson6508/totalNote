String gfg1 = String.join(" < ", "Four", "Five", "Six", "Seven"); 
System.out.println(gfg1);  // Four < Five < Six < Seven

String gfg2 = String.join("  ", "My", "name", "is", "Niraj", "Pandey"); 
System.out.println(gfg2);  // My name is Niraj Pandey
