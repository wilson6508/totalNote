CREATE TABLE weather_info ( 
	id int IDENTITY(1,1) NOT NULL,
	date date NOT NULL,
	city nvarchar(10) NOT NULL,
  	sunrise time,
	sunset time,
	high int,
	low int,
	aqi float,
	fx nvarchar(10),
	fl_max int,
	fl_min int,
	type nvarchar(10),
	notice text,
	update_time datetime NOT NULL,
  	PRIMARY KEY (id, date, city),
	INDEX sunrise (sunrise),
	INDEX sunset (sunset),
	INDEX high (high),
	INDEX low (low),
	INDEX aqi (aqi),
	INDEX fx (fx),
	INDEX fl_max (fl_max),
	INDEX fl_min (fl_min),
	INDEX update_time (update_time)
)
