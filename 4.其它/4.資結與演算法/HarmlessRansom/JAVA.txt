package test01;

import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;

public class Ex7 {
    public static void main(String[] args) {
        String text = "This is all the magazine text in the magazine";
        String target = "This magazine magazine apple";
//        String target = "This magazine magazine magazine";
        System.out.println(demo(text, target));
    }
    public static boolean demo(String text, String target) {
        // 計算text裡面單字的數量
        String[] textArr = text.split(" ");
        HashMap<String, Integer> hashMap = new HashMap<>();
        for (String str : textArr) {
            if (hashMap.containsKey(str)) {
                hashMap.put(str, hashMap.get(str) + 1);
            } else {
                hashMap.put(str, 1);
            }
        }
        // 遍歷target的單字
        String[] targetArr = target.split(" ");
        for (String str : targetArr) {
            if (hashMap.containsKey(str)) {
                hashMap.put(str, hashMap.get(str) - 1);
            } else {
                return false;
            }
        }
        List<Integer> list = new ArrayList<>(hashMap.values());
        return list.stream().allMatch(e -> e > -1);
    }
}
