function demo (numArray, targetSum) {
	const result = new Map();
	for (const [index, item] of numArray.entries()) {
		const check = targetSum - item;
		if (numArray.includes(check)) {
			const tempArr = [item, check].sort();
			if (!result.has(tempArr[0])) {
				result.set(tempArr[0], tempArr[1]);
			}		
		}
	}
	return [...result.entries()];
}

function demo (numArray, targetSum) {
	const mySet = new Set();
	for (const [index, item] of numArray.entries()) {
		const check = targetSum - item;
		if (numArray.includes(check)) {
			const tempArr = [item, check].sort();
			mySet.add(tempArr[0]);	
		}
	}
	return [...mySet].map(e => [e, targetSum - e]);
}
