package test01;

import java.util.Arrays;
import java.util.Date;
import java.util.List;

public class Ex4 {

    public static void main(String[] args) {
        List<Integer> numberList = Arrays.asList(1,3,5,7,9);
        System.out.println(demo(numberList, 9));
    }

    public static boolean demo(List<Integer> numberList, int target) {

        System.out.println(numberList.size() + " " + new Date());

        int middleIndex = (int) Math.floor(numberList.size() / 2.0);
        int middleVal = numberList.get(middleIndex);

        if (middleVal == target) {
            return true;
        }
        if (numberList.size() == 1) {
            return false;
        }
        if (target < middleVal) {
            return demo(numberList.subList(0, middleIndex), target);
        } else {
            return demo(numberList.subList(middleIndex + 1, numberList.size()), target);
        }

    }

}
