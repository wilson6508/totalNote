public class Ex7 {
    public static void main(String[] args) {
        int[] array = {1,3,5,7,9};
        System.out.println(demo(array, 2));
    }
    public static boolean demo(int[] nums, int target) {
        if (nums.length == 0) {
            return false;
        }
        boolean answer = false;
        int firstIdx = 0;
        int lastIdx = nums.length - 1;
        int middleIdx = -1;
        while (firstIdx <= lastIdx) {
            middleIdx = (firstIdx + lastIdx) / 2;
            if (target == nums[middleIdx]) {
                answer = true;
            }
            if (target < nums[middleIdx]) {
                lastIdx = middleIdx - 1;
            } else {
                firstIdx = middleIdx + 1;
            }
        }
        return answer;
    }
}
