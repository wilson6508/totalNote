package test01;

import java.util.*;
import java.util.stream.Collectors;

public class Ex7 {
    public static void main(String[] args) {
        System.out.println(isPalindrome("Madam, I'm Adam"));
        System.out.println(isPalindrome("Hello, I'm Adam"));
    }
    public static boolean isPalindrome(String str) {
        List<String> list = Arrays.asList(str.toLowerCase(Locale.ROOT).split(""));
        list = list.stream().filter(e -> e.matches("[a-z]")).collect(Collectors.toList());
        String one = String.join("", list);
        Collections.reverse(list);
        String two = String.join("", list);
        return one.equals(two);
    }
}
